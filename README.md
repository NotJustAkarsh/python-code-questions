## 💻 Sample Code

```python id="k3zq7e"
def is_palindrome(n):
    rev = 0
    temp = n
    
    while temp != 0:
        digit = temp % 10
        rev = rev * 10 + digit
        temp = temp // 10
        
    return rev == n


number = 121
print(is_palindrome(number))  # True
```
